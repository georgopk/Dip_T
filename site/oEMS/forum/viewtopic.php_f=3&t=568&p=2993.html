Hi All

I am new to OpenEMS and also to electrical field modelling in general. As a start, I made some adaptation to the Notch-Filter tutorial form the OpenEMS. This first script is supposed to model a simple metal path on a microchip. The end goal of mine is to obtain s-parameter files for on-chip inductors. However, my first script attempt already ended in an error which i don't understand. Here is my octave file:




%
% Calculation / RF Line
%

close all
clear
clc

%% setup the simulation %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
physical_constants;
unit = 1e-6; % specify everything in um
RFLINE_length = 500;
RFLINE_width = 4;
silicon_thickness = 254;
silicon_epr = 11.68;
silicon_kappa=4.3478e-04;
oxid_kappa=1e-16;
oxidLow_thickness=2.2650;
oxidLow_epr =3.0;
oxidHigh_thickness=13;
oxidHigh_epr =3.6;
air_thickness=2000;
air_epr = 1;
f_max = 60e9;
oxidLow_top=oxidLow_thickness+silicon_thickness;
oxidHigh_top=oxidLow_top+oxidHigh_thickness;
all_top=oxidHigh_top+air_thickness;

%% setup FDTD parameters & excitation function %%%%%%%%%%%%%%%%%%%%%%%%%%%%
FDTD = InitFDTD();
FDTD = SetGaussExcite( FDTD, f_max/2, f_max/2 );
BC = {'PML_8' 'PML_8' 'PML_8' 'PML_8' 'PML_8' 'PML_8'};
FDTD = SetBoundaryCond( FDTD, BC );

%% setup CSXCAD geometry & mesh %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
CSX = InitCSX();
resolution = c0/(f_max*sqrt(oxidHigh_epr))/unit /50; % resolution of lambda/50
mesh.x = SmoothMeshLines( [-4*RFLINE_length 4*RFLINE_length], resolution, 1.5 ,0 );
mesh.y = SmoothMeshLines( [0 RFLINE_width/2+[-resolution/3 +resolution/3*2]/4], resolution/4 , 1.5 ,0);
mesh.y = SmoothMeshLines( [-15*RFLINE_width -mesh.y mesh.y 15*RFLINE_width], resolution, 1.3 ,0);
mesh.z = SmoothMeshLines( [0 silicon_thickness oxidLow_top oxidHigh_top all_top], resolution );
CSX = DefineRectGrid( CSX, unit, mesh );

%% substrate
CSX = AddMaterial( CSX, 'Silicon' );
CSX = SetMaterialProperty( CSX, 'Silicon', 'Epsilon', silicon_epr, 'Kappa', 4.3478e-04);
start = [mesh.x(1), mesh.y(1), 0];
stop = [mesh.x(end), mesh.y(end), silicon_thickness];
CSX = AddBox( CSX, 'Silicon', 0, start, stop );

%% oxid low metal layers
CSX = AddMaterial( CSX, 'OxidLow' );
CSX = SetMaterialProperty( CSX, 'OxidLow', 'Epsilon', oxidLow_epr, 'Kappa', oxid_kappa);
start = [mesh.x(1), mesh.y(1), silicon_thickness];
stop = [mesh.x(end), mesh.y(end), oxidLow_top];
CSX = AddBox( CSX, 'OxidLow', 0, start, stop );

%% oxid upper metal layers
CSX = AddMaterial( CSX, 'OxidHigh' );
CSX = SetMaterialProperty( CSX, 'OxidHigh', 'Epsilon', oxidHigh_epr, 'Kappa', oxid_kappa);
start = [mesh.x(1), mesh.y(1), oxidLow_top];
stop = [mesh.x(end), mesh.y(end), oxidHigh_top];
CSX = AddBox( CSX, 'OxidLow', 0, start, stop );

%% Air upper metal layers
CSX = AddMaterial( CSX, 'Air' );
CSX = SetMaterialProperty( CSX, 'OxidHigh', 'Epsilon', 1);
start = [mesh.x(1), mesh.y(1), oxidHigh_top];
stop = [mesh.x(end), mesh.y(end), all_top];
CSX = AddBox( CSX, 'OxidLow', 0, start, stop );

%% MSL port
CSX = AddMetal( CSX, 'PEC' );
portstart = [ -RFLINE_length, -RFLINE_width/2, oxidLow_top];
portstop = [ 0, RFLINE_width/2, 0];
[CSX,port{1}] = AddMSLPort( CSX, 999, 1, 'PEC', portstart, portstop, 0, [0 0 -1], 'ExcitePort', true);

portstart = [RFLINE_length, -RFLINE_width/2, oxidLow_top];
portstop = [0 , RFLINE_width/2, 0];
[CSX,port{2}] = AddMSLPort( CSX, 999, 2, 'PEC', portstart, portstop, 0, [0 0 -1]);


%% write/show/run the openEMS compatible xml-file
Sim_Path = 'tmp';
Sim_CSX = 'rfline.xml';

[status, message, messageid] = rmdir( Sim_Path, 's' ); % clear previous directory
[status, message, messageid] = mkdir( Sim_Path ); % create empty simulation folder

WriteOpenEMS( [Sim_Path '/' Sim_CSX], FDTD, CSX );
CSXGeomPlot( 'tmp/rfline.xml' );
RunOpenEMS( Sim_Path, Sim_CSX );

%% post-processing
close all
f = linspace( 1e6, f_max, 1601 );
port = calcPort( port, Sim_Path, f, 'RefImpedance', 50);

s11 = port{1}.uf.ref./ port{1}.uf.inc;
s21 = port{2}.uf.ref./ port{1}.uf.inc;

plot(f/1e9,20*log10(abs(s11)),'k-','LineWidth',2);
hold on;
grid on;
plot(f/1e9,20*log10(abs(s21)),'r--','LineWidth',2);
legend('S_{11}','S_{21}');
ylabel('S-Parameter (dB)','FontSize',12);
xlabel('frequency (GHz) \rightarrow','FontSize',12);
ylim([-40 2]);
